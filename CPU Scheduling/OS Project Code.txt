  <script>
    let processes = [];

    function addProcess() {
      const pname = document.getElementById('pname').value.trim();
      const arrival = parseInt(document.getElementById('arrival').value);
      const burst = parseInt(document.getElementById('burst').value);
      const priority = parseInt(document.getElementById('priority').value);
      if (!pname || isNaN(arrival) || isNaN(burst)) return alert('Please fill all fields');
      processes.push({ pname, arrival, burst, priority });
      displayProcesses();
      document.getElementById('pname').value = '';
      document.getElementById('arrival').value = '';
      document.getElementById('burst').value = '';
      document.getElementById('priority').value = '';
    }

    function displayProcesses() {
      const tbody = document.querySelector('#processTable tbody');
      tbody.innerHTML = processes.map(p => `<tr><td>${p.pname}</td><td>${p.arrival}</td><td>${p.burst}</td><td>${p.priority||'-'}</td></tr>`).join('');
    }

    function runScheduling() {
      if (!processes.length) return alert('Add processes first!');
      const algo = document.getElementById('algorithm').value;
      const quantum = parseInt(document.getElementById('quantum').value);
      let result;
      switch(algo) {
        case 'fcfs': result = fcfs(); break;
        case 'sjf': result = sjf(); break;
        case 'srtf': result = srtf(); break;
        case 'rr': result = rr(quantum); break;
        case 'priority': result = prioritySchedule(); break;
      }
      displayResult(result);
    }

    function fcfs() {
      let t = 0, res = [];
      const sorted = [...processes].sort((a,b)=>a.arrival-b.arrival);
      for (let p of sorted) {
        if (t < p.arrival) t = p.arrival;
        let start = t;
        t += p.burst;
        res.push({p:p.pname, start, end:t});
      }
      return computeMetrics(res);
    }

    function sjf() {
      let t=0, completed=[], pending=[...processes];
      pending.sort((a,b)=>a.arrival-b.arrival);
      while(pending.length){
        let available = pending.filter(p=>p.arrival<=t);
        if(!available.length){t++;continue;}
        available.sort((a,b)=>a.burst-b.burst);
        let p = available[0];
        pending.splice(pending.indexOf(p),1);
        let start = t;
        t += p.burst;
        completed.push({p:p.pname,start,end:t});
      }
      return computeMetrics(completed);
    }

    function srtf() {
      let t=0, completed=[], pending=[...processes.map(p=>({...p, remaining:p.burst}))];
      while(pending.length){
        let available = pending.filter(p=>p.arrival<=t);
        if(!available.length){t++;continue;}
        available.sort((a,b)=>a.remaining-b.remaining);
        let current = available[0];
        current.remaining--;
        if(!completed.length || completed[completed.length-1].p!==current.pname)
          completed.push({p:current.pname,start:t});
        t++;
        if(current.remaining===0){
          current.end=t;
          pending.splice(pending.indexOf(current),1);
          completed[completed.length-1].end=t;
        }
      }
      return computeMetrics(completed);
    }

    function rr(q) {
      let queue=[...processes.map(p=>({...p, remaining:p.burst}))].sort((a,b)=>a.arrival-b.arrival);
      let t=queue[0].arrival, completed=[];
      while(queue.length){
        let p=queue.shift();
        if(t<p.arrival) t=p.arrival;
        let exec=Math.min(q,p.remaining);
        let start=t; t+=exec; p.remaining-=exec;
        completed.push({p:p.pname,start,end:t});
        if(p.remaining>0) queue.push(p);
      }
      return computeMetrics(completed);
    }

    function prioritySchedule() {
      let t=0, completed=[], pending=[...processes];
      while(pending.length){
        let available=pending.filter(p=>p.arrival<=t);
        if(!available.length){t++;continue;}
        available.sort((a,b)=>a.priority-b.priority);
        let p=available[0];
        pending.splice(pending.indexOf(p),1);
        let start=t; t+=p.burst;
        completed.push({p:p.pname,start,end:t});
      }
      return computeMetrics(completed);
    }

    function computeMetrics(gantt){
      const map={};
      gantt.forEach(seg=>{
        if(!map[seg.p]) map[seg.p]={start:seg.start,end:seg.end};
        else map[seg.p].end=seg.end;
      });
      const results=[];
      for(let p of processes){
        let g=map[p.pname];
        let completion=g.end;
        let turnaround=completion-p.arrival;
        let waiting=turnaround-p.burst;
        results.push({p:p.pname,completion,turnaround,waiting});
      }
      return {gantt,results};
    }

    function displayResult({gantt,results}){
      const ganttDiv=document.getElementById('ganttChart');
      ganttDiv.innerHTML='';
      gantt.forEach(seg=>{
        const div=document.createElement('div');
        div.className='bar';
        div.textContent=seg.p;
        ganttDiv.appendChild(div);
      });
      const tbody=document.querySelector('#resultTable tbody');
      tbody.innerHTML=results.map(r=>`<tr><td>${r.p}</td><td>${r.completion}</td><td>${r.turnaround}</td><td>${r.waiting}</td></tr>`).join('');
      const avgW=(results.reduce((a,b)=>a+b.waiting,0)/results.length).toFixed(2);
      const avgT=(results.reduce((a,b)=>a+b.turnaround,0)/results.length).toFixed(2);
      document.getElementById('summary').innerHTML=`Avg Waiting: ${avgW} | Avg Turnaround: ${avgT}`;
    }

    function smartSuggest(){
      const algos=['fcfs','sjf','srtf','rr','priority'];
      let best={algo:'',avgW:Infinity};
      for(let a of algos){
        let q=a==='rr'?2:0; // default quantum
        let res;
        try{switch(a){
          case 'fcfs':res=fcfs();break;
          case 'sjf':res=sjf();break;
          case 'srtf':res=srtf();break;
          case 'rr':res=rr(q);break;
          case 'priority':res=prioritySchedule();break;
        }}catch(e){continue;}
        let avgW=res.results.reduce((x,y)=>x+y.waiting,0)/res.results.length;
        if(avgW<best.avgW) best={algo:a,avgW};
      }
      alert(`Smart Suggestion: Best Algorithm = ${best.algo.toUpperCase()} (Avg Waiting: ${best.avgW.toFixed(2)})`);
    }
